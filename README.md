# configuration setting
```javascript
import Phaser from './scenes/phamport.js'
import {SceneMain} from './scenes/gameScene.js'

let config = {
  type: Phaser.AUTO,
  width: 800,
  height: 600,
  parent: 'phaser-game',
  scene: [SceneMain]
};

var game = new Phaser.Game(config);
```
